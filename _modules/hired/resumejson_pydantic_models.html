class Iso8601(
    RootModel[
        constr(
            pattern=r"^([1-2][0-9]{3}-[0-1][0-9]-[0-3][0-9]|[1-2][0-9]{3}-[0-1][0-9]|[1-2][0-9]{3})$"
        )
    ]
):
    root: constr(
        pattern=r"^([1-2][0-9]{3}-[0-1][0-9]-[0-3][0-9]|[1-2][0-9]{3}-[0-1][0-9]|[1-2][0-9]{3})$"
    ) = Field(..., description="e.g. 2014-06-29")
